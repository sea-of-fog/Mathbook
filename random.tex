\begin{thm}[Duality law for Boolean algebra]
\label{BooleanAlgebraDuality}
\hypertarget{BooleanAlgebraDuality}
Every identity of boolean algebra is has a dual identity, i.e. one obtained by interchanging \( 0 \) with \( 1 \) and \( \wedge \) with \( \vee \).
\ankimark
\end{thm}
\begin{proof}
Negation!
\end{proof}
